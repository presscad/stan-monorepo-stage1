7 void compute_width_and_precision(double value, int sig_figs, int& width, int& precision) {
8 
9  double abs_value = std::fabs(value);
10 
11  if (value == 0) {
12  width = sig_figs;
13  precision = sig_figs;
14  }
15  else if (abs_value >= 1) {
16  int int_part = std::ceil(log10(abs_value) + 1e-6);
17  width = int_part >= sig_figs ? int_part : sig_figs + 1;
18  precision = int_part >= sig_figs ? 0 : sig_figs - int_part;
19  }
20  else {
21  int frac_part = std::fabs(std::floor(log10(abs_value)));
22  width = 1 + frac_part + sig_figs;
23  precision = frac_part + sig_figs - 1;
24  }
25 
26  if (value < 0) ++width;
27 
28 }

162 int matrix_index(std::vector<int>& index, const std::vector<int>& dims) {
163  int offset = 0;
164  int prod = 1;
165  for (int i = 0; i < dims.size(); i++) {
166  offset += (index[i]-1) * prod;
167  prod *= dims[i];
168  }
