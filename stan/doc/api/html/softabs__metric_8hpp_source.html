114  void update_metric(softabs_point& z,
115  callbacks::writer& info_writer,
116  callbacks::writer& error_writer) {
117  // Compute the Hessian
118  stan::math::hessian<double>(softabs_fun<Model>(this->model_, 0),
119  z.q, z.V, z.g, z.hessian);
120 
121  z.V = -z.V;
122  z.g = -z.g;
123  z.hessian = -z.hessian;
124 
125  // Compute the eigen decomposition of the Hessian,
126  // then perform the SoftAbs transformation
127  z.eigen_deco.compute(z.hessian);
128 
129  for (idx_t i = 0; i < z.q.size(); ++i) {
130  double lambda = z.eigen_deco.eigenvalues()(i);
131  double alpha_lambda = z.alpha * lambda;
132 
133  double softabs_lambda = 0;
134 
135  // Thresholds defined such that the approximation
136  // error is on the same order of double precision
137  if (std::fabs(alpha_lambda) < lower_softabs_thresh) {
138  softabs_lambda = (1.0
139  + (1.0 / 3.0) * alpha_lambda * alpha_lambda)
140  / z.alpha;
141  } else if (std::fabs(alpha_lambda) > upper_softabs_thresh) {
142  softabs_lambda = std::fabs(lambda);
143  } else {
144  softabs_lambda = lambda / std::tanh(alpha_lambda);
145  }
146 
147  z.softabs_lambda(i) = softabs_lambda;
148  z.softabs_lambda_inv(i) = 1.0 / softabs_lambda;
149  }
150 
151  // Compute the log determinant of the metric
152  z.log_det_metric = 0;
153  for (idx_t i = 0; i < z.q.size(); ++i)
154  z.log_det_metric += std::log(z.softabs_lambda(i));
155  }

157  void update_metric_gradient(softabs_point& z,
158  callbacks::writer& info_writer,
159  callbacks::writer& error_writer) {
160  // Compute the pseudo-Jacobian of the SoftAbs transform
161  for (idx_t i = 0; i < z.q.size(); ++i) {
162  for (idx_t j = 0; j <= i; ++j) {
163  double delta = z.eigen_deco.eigenvalues()(i)
164  - z.eigen_deco.eigenvalues()(j);
165 
166  if (std::fabs(delta) < jacobian_thresh) {
167  double lambda = z.eigen_deco.eigenvalues()(i);
168  double alpha_lambda = z.alpha * lambda;
169 
170  // Thresholds defined such that the approximation
171  // error is on the same order of double precision
172  if (std::fabs(alpha_lambda) < lower_softabs_thresh) {
173  z.pseudo_j(i, j) = (2.0 / 3.0) * alpha_lambda
174  * (1.0 - (2.0 / 15.0)
175  * alpha_lambda * alpha_lambda);
176  } else if (std::fabs(alpha_lambda) > upper_softabs_thresh) {
177  z.pseudo_j(i, j) = lambda > 0 ? 1 : -1;
178  } else {
179  double sdx = std::sinh(alpha_lambda) / lambda;
180  z.pseudo_j(i, j) = (z.softabs_lambda(i)
181  - z.alpha / (sdx * sdx) ) / lambda;
182  }
183  } else {
184  z.pseudo_j(i, j) = (z.softabs_lambda(i)
185  - z.softabs_lambda(j) ) / delta;
186  }
187  }
188  }
189  }
