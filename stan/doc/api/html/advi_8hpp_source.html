98  double calc_ELBO(const Q& variational,
99  callbacks::writer& message_writer)
100  const {
101  static const char* function =
102  "stan::variational::advi::calc_ELBO";
103 
104  double elbo = 0.0;
105  int dim = variational.dimension();
106  Eigen::VectorXd zeta(dim);
107 
108  int n_dropped_evaluations = 0;
109  for (int i = 0; i < n_monte_carlo_elbo_;) {
110  variational.sample(rng_, zeta);
111  try {
112  std::stringstream ss;
113  double log_prob = model_.template log_prob<false, true>(zeta, &ss);
114  if (ss.str().length() > 0)
115  message_writer(ss.str());
116  stan::math::check_finite(function, "log_prob", log_prob);
117  elbo += log_prob;
118  ++i;
119  } catch (const std::domain_error& e) {
120  ++n_dropped_evaluations;
121  if (n_dropped_evaluations >= n_monte_carlo_elbo_) {
122  const char* name = "The number of dropped evaluations";
123  const char* msg1 = "has reached its maximum amount (";
124  const char* msg2 = "). Your model may be either severely "
125  "ill-conditioned or misspecified.";
126  stan::math::domain_error(function, name, n_monte_carlo_elbo_,
127  msg1, msg2);
128  }
129  }
130  }
131  elbo /= n_monte_carlo_elbo_;
132  elbo += variational.entropy();

208  }
209 
210  // Variational family to store gradients
211  Q elbo_grad = Q(model_.num_params_r());
212 
213  // Adaptive step-size sequence
214  Q history_grad_squared = Q(model_.num_params_r());
215  double tau = 1.0;
216  double pre_factor = 0.9;
217  double post_factor = 0.1;
218

222

228 
229  int print_progress_m;
230  for (int iter_tune = 1; iter_tune <= adapt_iterations; ++iter_tune) {
231  print_progress_m = eta_sequence_index
232  * adapt_iterations + iter_tune;
233  variational
234  ::print_progress(print_progress_m, 0,
235  adapt_iterations * eta_sequence_size,
236  adapt_iterations, true, "", "", message_writer);
237 
238  // (ROBUST) Compute gradient of ELBO. It's OK if it diverges.
239  // We'll try a smaller eta.
240  try {
241  calc_ELBO_grad(variational, elbo_grad, message_writer);
242  } catch (const std::domain_error& e) {
243  elbo_grad.set_to_zero();
244  }
245 
246  // Update step-size
247  if (iter_tune == 1) {
248  history_grad_squared += elbo_grad.square();
249  } else {
250  history_grad_squared = pre_factor * history_grad_squared
251  + post_factor * elbo_grad.square();
252  }
253  eta_scaled = eta / sqrt(static_cast<double>(iter_tune));
254  // Stochastic gradient update
255  variational += eta_scaled * elbo_grad
256  / (tau + history_grad_squared.sqrt());
257  }
258 
259  // (ROBUST) Compute ELBO. It's OK if it has diverged.
260  try {
261  elbo = calc_ELBO(variational, message_writer);
262  } catch (const std::domain_error& e) {
263  elbo = -std::numeric_limits<double>::max();
264  }
265

267  // (1) ELBO at current eta is worse than the best ELBO
268  // (2) the best ELBO hasn't gotten worse than the initial ELBO
269  if (elbo < elbo_best && elbo_best > elbo_init) {
270  std::stringstream ss;
271  ss << "Success!"
272  << " Found best value [eta = " << eta_best
273  << "]";
274  if (eta_sequence_index < eta_sequence_size - 1)
275  ss << (" earlier than expected.");
276  else
277  ss << ".";
278  message_writer(ss.str());
279  message_writer();
280  do_more_tuning = false;
281  } else {
282  if (eta_sequence_index < eta_sequence_size - 1) {
283  // Reset
284  elbo_best = elbo;
285  eta_best = eta;
286  } else {
287  // No more eta values to try, so use current eta if it
288  // didn't diverge or fail if it did diverge
289  if (elbo > elbo_init) {
290  std::stringstream ss;
291  ss << "Success!"
292  << " Found best value [eta = " << eta_best
293  << "].";
294  message_writer(ss.str());
295  message_writer();
296  eta_best = eta;
297  do_more_tuning = false;
298  } else {
299  const char* name = "All proposed step-sizes";
300  const char* msg1 = "failed. Your model may be either "
301  "severely ill-conditioned or misspecified.";
302  stan::math::domain_error(function, name, "", msg1);
303  }
304  }
305  // Reset
306  history_grad_squared.set_to_zero();
307  }
308  ++eta_sequence_index;
309  variational = Q(cont_params_);
310  }
311  return eta_best;
